import { MetadataRoute } from 'next';

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      // We can disallow any internal admin or CRM routing later
      disallow: '/private/',
    },
    sitemap: 'https://www.omirawellness.com/sitemap.xml',
  };
}
